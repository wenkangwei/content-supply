# Content Supply Web - v1.0 数据协议

> 版本: v1.0 | 日期: 2026-05-08 | 对接后端: content-supply @ localhost:8010

---

## 一、通信规范

| 项目 | 值 |
|------|------|
| 协议 | HTTP/1.1 |
| 数据格式 | JSON (`application/json`) |
| 字符编码 | UTF-8 |
| 时间格式 | ISO 8601 (`2026-05-08T10:30:00`) |
| 分页风格 | `skip` + `limit` (Feeds/Hot) / `page` + `page_size` (Items) |
| 认证 | 无（本地部署） |

### 通用响应

```typescript
// 成功（单条）
interface SuccessResponse<T> {
  data: T;
  message: string;
}

// 列表（分页）
interface PaginatedResponse<T> {
  items: T[];
  total: number;
  page: number;
  page_size: number;
}

// 错误
interface ErrorResponse {
  detail: string;
}
```

### HTTP 状态码

| 状态码 | 含义 | 场景 |
|--------|------|------|
| 200 | 成功 | GET/PUT 成功 |
| 201 | 创建成功 | POST 创建资源 |
| 204 | 无内容 | DELETE 成功 |
| 400 | 参数错误 | 请求体验证失败 |
| 404 | 不存在 | 资源ID不存在 |
| 409 | 冲突 | Feed URL重复 |
| 500 | 服务器错误 | 后端异常 |

## 二、TypeScript 类型定义

### 2.1 Feed 模块

```typescript
// source_type 枚举
type SourceType = 'rss' | 'atom' | 'web' | 'hot_search';

// Feed 状态
type FeedStatus = 'active' | 'paused' | 'error';

// Feed 完整类型
interface Feed {
  id: number;
  name: string;
  url: string;
  source_type: SourceType;
  category?: string;
  poll_interval: number;         // 轮询间隔（秒），默认 1800
  status: FeedStatus;
  last_fetched_at?: string;      // ISO 8601
  last_error?: string;
  error_count: number;
  created_at: string;
  updated_at?: string;
}

// 创建 Feed
interface FeedCreate {
  name: string;                  // 必填
  url: string;                   // 必填，唯一
  source_type: SourceType;       // 必填
  category?: string;
  poll_interval?: number;        // 默认 1800
}

// 更新 Feed（部分更新）
interface FeedUpdate {
  name?: string;
  url?: string;
  source_type?: SourceType;
  category?: string;
  poll_interval?: number;
}

// 切换状态响应
interface FeedToggleResponse {
  id: number;
  name: string;
  status: FeedStatus;
  message: string;
}
```

### 2.2 Item 模块

```typescript
// Item 来源类型
type ItemSourceType = 'rss' | 'web' | 'hot_keyword' | 'manual' | 'jimeng';

// Item 状态
type ItemStatus = 'draft' | 'published' | 'archived';

// 内容类型
type ContentType = 'article' | 'video' | 'post';

// Item 完整类型
interface Item {
  id: string;                    // 内容哈希ID（64字符）
  title: string;
  summary?: string;
  content?: string;              // 全文（可能被改写）
  original_content?: string;     // 改写前原文
  url: string;
  image_url?: string;
  author?: string;
  source_name?: string;
  source_type: ItemSourceType;
  feed_id?: number;
  hot_keyword_id?: number;
  content_type?: ContentType;
  category?: string;
  tags?: string;                 // 逗号分隔
  quality_score: number;         // 0.0 ~ 1.0
  content_hash?: string;
  is_rewritten: boolean;
  rewrite_task_id?: number;
  exposure_count: number;
  click_count: number;
  status: ItemStatus;
  published_at?: string;
  created_at: string;
  updated_at?: string;
}

// 列表查询参数
interface ItemListParams {
  skip?: number;                 // 默认 0
  limit?: number;                // 默认 20
  source_type?: ItemSourceType;
  category?: string;
  status?: ItemStatus;
}

// 搜索请求
interface ItemSearchParams {
  query: string;                 // 必填
  page_size?: number;            // 默认 20
  page?: number;                 // 默认 1
  source_type?: ItemSourceType;
  category?: string;
}

// 状态更新响应
interface ItemStatusResponse {
  message: string;
}
```

### 2.3 Crawl 模块

```typescript
// 任务类型
type TaskType = 'rss' | 'web' | 'manual' | 'hot_keyword';

// 任务状态
type TaskStatus = 'pending' | 'running' | 'done' | 'failed';

// 抓取任务
interface CrawlTask {
  id: number;
  feed_id?: number;
  hot_keyword_id?: number;
  url: string;
  task_type: TaskType;
  status: TaskStatus;
  items_found: number;
  items_new: number;
  error_message?: string;
  started_at?: string;
  finished_at?: string;
  created_at: string;
}

// URL 爬取请求
interface CrawlUrlRequest {
  url: string;                   // 必填
  category?: string;
}

// URL 爬取响应
interface CrawlUrlResponse {
  task: CrawlTask;
  item?: {
    title: string;
    url: string;
    content?: string;
    author?: string;
    image_url?: string;
    published_at?: string;
    source_type: string;
    tags?: string[];
  };
}
```

## 四、前端状态管理协议

### 4.1 Zustand Store 结构

```typescript
// useAppStore - 全局应用状态
interface AppStore {
  theme: 'dark' | 'light';
  sidebarCollapsed: boolean;
  apiConnected: boolean;
  toggleTheme: () => void;
  toggleSidebar: () => void;
  setApiConnected: (v: boolean) => void;
}

// useItemStore - 内容状态
interface ItemStore {
  items: Item[];
  total: number;
  loading: boolean;
  currentFilters: ItemListParams;
  searchQuery: string;
  fetchItems: (params?: ItemListParams) => Promise<void>;
  searchItems: (query: string) => Promise<void>;
  setFilters: (filters: Partial<ItemListParams>) => void;
}
```

### 4.2 缓存策略

| 数据 | 缓存方式 | TTL |
|------|----------|-----|
| Feed 列表 | Zustand store | 页面级（切换页面刷新） |
| Item 列表 | Zustand store | 不缓存（每次重新请求） |
| 仪表板统计 | 组件内 state | 5分钟 |
| 热搜词 | Zustand store | 10分钟 |
| 清理策略 | 组件内 state | 30分钟 |
